

/* Create Tables */

-- 게시판
CREATE TABLE BULLETIN_BOARD
(
	-- 게시판코드
	BBI_CODE char(1) NOT NULL,
	-- 게시물번호
	BB_NUM varchar2(5) NOT NULL,
	-- 회원코드
	MBR_CODE varchar2(5) NOT NULL,
	-- 제품코드
	PRD_CODE varchar2(4),
	-- 제목
	BB_TITL varchar2(100) NOT NULL,
	-- 내용
	BB_CN varchar2(3000) NOT NULL,
	-- 조회수
	BB_RDCNT number(5) NOT NULL,
	-- 등록날짜
	BB_REGIST_DATE varchar2(14) NOT NULL,
	-- 비밀번호
	BB_PWD varchar2(4),
	-- 게시물공개여부
	BB_OTHBC_AT char(1) NOT NULL,
	PRIMARY KEY (BBI_CODE, BB_NUM)
);


-- 게시판댓글
CREATE TABLE BULLETIN_BOARD_ANSWER
(
	-- 댓글번호
	ANSWER_NUM varchar2(5) NOT NULL,
	-- 게시판코드
	BBI_CODE char(1) NOT NULL,
	-- 게시물번호
	BB_NUM varchar2(5) NOT NULL,
	-- 회원코드
	MBR_CODE varchar2(5) NOT NULL,
	-- 작성날짜
	ANSWER_DATE varchar2(14) NOT NULL,
	-- 내용
	ANSWER_CN varchar2(600) NOT NULL,
	PRIMARY KEY (ANSWER_NUM)
);


-- 게시판사진
CREATE TABLE BULLETIN_BOARD_IMAGE
(
	-- 게시판코드
	BBI_CODE char(1) NOT NULL,
	-- 게시물번호
	BB_NUM varchar2(5) NOT NULL,
	-- 이미지코드
	BB_IMAGE_CODE varchar2(2) NOT NULL,
	-- 이미지주소
	BB_IMAGE_URL varchar2(100) NOT NULL,
	PRIMARY KEY (BBI_CODE, BB_NUM, BB_IMAGE_CODE)
);


-- 게시물정보
CREATE TABLE BULLETIN_BOARD_INFORMATION
(
	-- 게시판코드
	BBI_CODE char(1) NOT NULL,
	-- 게시판명
	BBI_NAME varchar2(9) NOT NULL UNIQUE,
	PRIMARY KEY (BBI_CODE)
);


-- 장바구니
CREATE TABLE CART
(
	-- 회원코드
	MBR_CODE varchar2(5) NOT NULL,
	-- 제품코드
	PRD_CODE varchar2(4) NOT NULL,
	-- 수량
	QY number(3),
	PRIMARY KEY (MBR_CODE, PRD_CODE)
);


-- 공통
CREATE TABLE COMMON
(
	-- 공통코드
	COMMON_CODE varchar2(4) NOT NULL,
	-- 공통코드명
	COMMON_NAME varchar2(30) NOT NULL,
	PRIMARY KEY (COMMON_CODE)
);


-- 배송
CREATE TABLE DELIVERY
(
	-- 배송코드
	DLVY_CODE varchar2(6) NOT NULL,
	-- 수령자명
	RECPT_NAME varchar2(21) NOT NULL,
	-- 연락처
	DLVY_CTTPC varchar2(15) NOT NULL,
	-- 우편번호
	DLVY_ZIP varchar2(6),
	-- 상세주소
	DLVY_DETAIL_ADRES varchar2(300),
	PRIMARY KEY (DLVY_CODE)
);


-- 회원
CREATE TABLE DMEMBER
(
	-- 회원코드
	MBR_CODE varchar2(5) NOT NULL,
	-- 아이디
	MBR_ID varchar2(12) NOT NULL UNIQUE,
	-- 비밀번호
	MBR_PWD varchar2(12) NOT NULL,
	-- 성명
	MBR_NAME varchar2(21) NOT NULL,
	-- 생년월일
	MBR_BRTHDY varchar2(8) NOT NULL,
	-- 성별
	MBR_SEX char(1) NOT NULL,
	-- 이메일
	MBR_EMAIL varchar2(30),
	-- 전화번호
	MBR_TELNO varchar2(15),
	-- 등급
	MBR_GRADE varchar2(4) NOT NULL,
	-- 가입날짜
	SBSCRB_DATE varchar2(8) NOT NULL,
	-- 학년반
	SCHUL_GRADE_CLASS varchar2(4),
	PRIMARY KEY (MBR_CODE)
);


-- 주문
CREATE TABLE DORDER
(
	-- 주문코드
	ORDER_CODE varchar2(6) NOT NULL,
	-- 회원코드
	MBR_CODE varchar2(5) NOT NULL,
	-- 배송코드
	DLVY_CODE varchar2(6) NOT NULL,
	-- 날짜
	ORDER_DATE varchar2(14) NOT NULL,
	-- 결제방법
	SETLE_MTH varchar2(4) NOT NULL,
	-- 수령방법
	RECPT_MTH varchar2(4) NOT NULL,
	-- 진행현황
	PROGRS_STTUS varchar2(4) NOT NULL,
	-- 택배사명
	HDRL_NAME varchar2(30),
	-- 운송장번호
	WAYBIL_NUM varchar2(20),
	-- 요구사항
	DEMAND varchar2(300),
	-- 총금액
	TOTAL_PC number(6) NOT NULL,
	PRIMARY KEY (ORDER_CODE)
);


-- 주문상세
CREATE TABLE ORDER_DETAIL
(
	-- 주문코드
	ORDER_CODE varchar2(6) NOT NULL,
	-- 제품코드
	PRD_CODE varchar2(4) NOT NULL,
	-- 상세주문수량
	DETAIL_ORDER_QY number(3) NOT NULL,
	PRIMARY KEY (ORDER_CODE, PRD_CODE)
);


-- 제품
CREATE TABLE PRODUCT
(
	-- 제품코드
	PRD_CODE varchar2(4) NOT NULL,
	-- 분류코드
	PCL_CODE varchar2(4) NOT NULL,
	-- 제품명
	PRD_NAME varchar2(60) NOT NULL,
	-- 가격
	PRD_PC number(6) NOT NULL,
	-- 재고
	PRD_INVNTRY number(3) NOT NULL,
	-- 내용
	PRD_CN varchar2(3000),
	-- 등록날짜
	PRD_REGIST_DATE varchar2(8) NOT NULL,
	-- 추천여부
	PRD_RECOMEND_AT char(1) NOT NULL,
	-- 추천수
	PRD_RECOMEND_CNT number(5) NOT NULL,
	-- 제품구분
	PRD_SE varchar2(4) NOT NULL,
	PRIMARY KEY (PRD_CODE)
);


-- 상품분류
CREATE TABLE PRODUCT_CLASSIFICATION
(
	-- 분류코드
	PCL_CODE varchar2(4) NOT NULL,
	-- 분류명
	PCL_NAME varchar2(30) NOT NULL,
	PRIMARY KEY (PCL_CODE)
);


-- 제품사진
CREATE TABLE PRODUCT_IMAGE
(
	-- 제품코드
	PRD_CODE varchar2(4) NOT NULL,
	-- 이미지코드
	PRD_IMAGE_CODE varchar2(2) NOT NULL,
	-- 이미지주소
	PRD_IMAGE_URL varchar2(100) NOT NULL,
	PRIMARY KEY (PRD_CODE, PRD_IMAGE_CODE)
);


-- 스쿨존
CREATE TABLE SCHOOL_ZONE
(
	-- 스쿨존코드
	SCHUL_CODE varchar2(4) NOT NULL,
	-- 알림글
	SCHUL_MSG varchar2(600),
	PRIMARY KEY (SCHUL_CODE)
);
